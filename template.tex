% Styles for section dividers and job header lines. Keep it simple.
\newcommand{\sectionstyle}[1]{#1}

% Finagle with the floating icons and contact info under the name

% Expect subsections like "Foo @ Bar". Left-align Foo and right-align Bar. Also
% format arrows "->" nicely.
\newcommand{\fmtsection}[1]{{%
  \noexpandarg % suppress expansions made by xstring
  \StrSubstitute{#1}{\%}{\%25}[\x]% first step
  \expandafter\StrSubstitute\expandafter{\x}{-\textgreater}{\textrightarrow}[\x]%
  \expandafter\StrSubstitute\expandafter{\x}{@}{\hfill}[\x]%
  \x}}
\renewcommand{\subsection}[1]{\jobstyle{\fmtsection{#1}}}

\newcommand{\tag}[2]{\includegraphics[height=8px]{#1}\hspace{0.5em}\fancy{#2}}

\let\olditemize\itemize

\renewenvironment{itemize}{\vspace{-10pt}\begin{olditemize}}{\end{olditemize}}

\renewcommand{\rule}[2]{{\color{Purple}\sout{\hfill}}}

\titlespacing{\section}{0pt}{\parskip}{-\parskip}

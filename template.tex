% Fancy font
\usepackage{ocr}
\newcommand{\fancy}[1]{\small{\ocrfamily{#1}}}

% Finagle with the floating icons and contact info under the name
\newcommand{\tag}[2]{\includegraphics[height=8px]{#1}\hspace{0.5em}\fancy{#2}}
\newcommand{\emailtag}{\tag{icons/envelope-regular.pdf}{\href{mailto:$email$}{$email$}}}
\newcommand{\webtag}{\tag{icons/link-solid.pdf}{\href{$url$}{$url$}}}
\newcommand{\phonetag}{\tag{icons/phone-solid.pdf}{\href{tel:$phone$}{$phone$}}}
\newcommand{\citytag}{\tag{icons/map-marker-alt-solid.pdf}{$location$}}

% Redefine itemize to have a slight upward shift
\let\olditemize\itemize
\let\endolditemize\enditemize
\renewenvironment{itemize}{\vspace{-15pt}\begin{olditemize}}{\end{olditemize}\vspace{-7pt}}

% Pandoc throws \tightlist in there automatically, so we'd better define it
\providecommand{\tightlist}{\setlength{\itemsep}{3pt}\setlength{\parskip}{0pt}}

% Horizontal line across section breaks. The argument is the name of the
% section, just in case we want to get that back in there.
\renewcommand{\section}[1]{{\color{Purple}\sout{\hfill}}}

% Expect subsections like "Foo @ Bar". Left-align Foo and right-align Bar.
\setsepchar{@}
\renewcommand{\subsection}[1]{\readlist*\elts{#1@}\fancy{\elts[1]}\hfill\fancy{\elts[2]}}
